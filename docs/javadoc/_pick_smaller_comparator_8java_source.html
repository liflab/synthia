26  * A variant of {@link PickIf} that selects an element if it is smaller than

33  * @ingroup API
34  */

47  /**

51  * @param c The comparator used to compare the reference to candidate values
52  */
53  public PickSmallerComparator(Picker<? extends T> source, T o, Comparator<T> c)

63  PickSmallerComparator<T> ps = new PickSmallerComparator<T>(m_picker.duplicate(with_state), m_reference, m_comparator);
64  if (with_state)

72  PickSmallerComparator<T> ps = new PickSmallerComparator<T>(m_picker, o, c);
73  ps.m_maxIteration = m_maxIteration;
74  return ps;
75  }
76 
77  @Override
